JAVA中方法详解

public class Test001 {

 //成员变量
int a = 0;

//静态变量
static int b = 1;

//成员方法
public void m1(){

}

//静态方法
public static void m2(){

}

//成员内部类
class Inner{

    public void innertest(){//成员方法
        int c = a;
        c = b;
        m1();
        m2();
    }

// public static void innertest0(){//不能定义静态方法
//
// }
}

//静态内部类
static class Inner0{
    public void inner0test(){//成员方法

// int c = a;Cannot make a static reference to the non-static method m1() from the type Test001
int c = b;
// m1();Cannot make a static reference to the non-static method m1() from the type Test001
m2();
}

public static void inner0test0(){

// int c = a;
int c = b;
// m1();
m2();
}
}

public void m3(){

    int d = 4;

    //局部内部类
    class Inner3{

        public void inner3test(){//成员方法

// System.out.println(d);//不能这样用，除非d为final，Cannot refer to a non-final variable d inside an inner class defined in a different method

}

// public static void inner3test0(){//不能用静态方法
//
// }
}

    //在这里调用
    //newInner对象
    Inner3 inner3 = new Inner3();
    inner3.inner3test();

}

public static void main(String[] args) {

    System.out.println("begin");

    //调用方法：

    //调用成员内部类：

// Inner in = new Inner();//这种方式不行
// Test001.Inner = new Test001.Inner();这样也不行
// Test001.Inner.innertest();//这样也不行
//正确方法：
Test001 t = new Test001();//创建外部类对象
Inner in = t.new Inner();//创建内部类对象
in.innertest();//调用

//调用静态内部类：
Inner0 in0 = new Inner0();//直接new
in0.inner0test();//调用成员
in0.inner0test0();//调用静态
Inner0 in01 =new Test001.Inner0();//也可以这样new
Test001.Inner0.inner0test0();//也可以这样调用静态内部类的静态方法

// Test001.Inner0.inner0test();不可以这样调用静态内部类的成员方法
// Inner in0

//调用局部内部类：
//必须在Test001中调用（main之外）

//使用匿名内部类：

// 如果我想调用logout方法，有两种选择：
//1.创建接口fuzhu的实现类
Fuzhu shixian = new Shixian();
shixian.logout();
//2.使用内部类
new Fuzhu(){
public void logout() {
}
};
}
}
